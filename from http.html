from http.server import HTTPServer, BaseHTTPRequestHandler content = """ My webserver

Top five Revenue generating Software Companies

  1. Microsoft
  2. Oracle
  3. Salesforce
  4. IBM
  5. Intuit
""" class myhandler(BaseHTTPRequestHandler): def do_GET(self): print("request received") self.send_response(200) self.send_header('content-type', 'text/html; charset=utf-8') self.end_headers() self.wfile.write(content.encode()) server_address = ('',8004) httpd = HTTPServer(server_address,myhandler) print("my webserver is running...") httpd.serve_forever()
